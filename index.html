Conceitos Básicos de JavaScript

Primeiramente baixe e instale o Nodejs

Variáveis

As variáveis são usadas para armazenar dados que podem ser manipulados e reutilizados. Em JavaScript, podemos declarar variáveis usando var, let e const.

var nome = "João";
        let idade = 25;
        const cidade = "São Paulo";

Tipos de Dados

JavaScript possui vários tipos de dados, incluindo:

  • String: Cadeia de caracteres. Ex: "Olá, mundo!"
  • Number: Números. Ex: 42, 3.14
  • Boolean: Verdadeiro ou falso. Ex: true, false
  • Null: Valor nulo. Ex: null
  • Undefined: Variável não inicializada. Ex: undefined
  • Object: Estruturas complexas de dados. Ex: {nome: "João", idade: 25}
  • Array: Lista de valores. Ex: [1, 2, 3, 4, 5]

Operadores

Alguns operadores em JavaScript:

  • Aritméticos: +, -, *, /, %
  • Comparação: ==, ===, !=, !==, >, <, >=, <=
  • Lógicos: &&, ||, !

Estruturas de Controle

Condicionais

let hora = 10;

        if (hora < 12) {
            console.log("Bom dia!");
        } else if (hora < 18) {
            console.log("Boa tarde!");
        } else {
            console.log("Boa noite!");
        }

Laços

for (let i = 0; i < 5; i++) {
            console.log(i);
        }

Funções

Definindo Funções

function saudacao(nome) {
            return "Olá, " + nome + "!";
        }

        console.log(saudacao("Maria"));

Parâmetros e Argumentos

function soma(a, b) {
            return a + b;
        }

        console.log(soma(5, 3));

Funções Anônimas e Arrow Functions

let saudacao = function(nome) {
            return "Olá, " + nome + "!";
        };

        let saudacaoArrow = (nome) => "Olá, " + nome + "!";

        console.log(saudacao("José"));
        console.log(saudacaoArrow("Ana"));

Objetos e Arrays

Criando e Manipulando Objetos

let pessoa = {
            nome: "Carlos",
            idade: 30,
            cidade: "Rio de Janeiro"
        };

        console.log(pessoa.nome);
        pessoa.idade = 31;
        console.log(pessoa);

Arrays e Métodos de Arrays

let numeros = [1, 2, 3, 4, 5];

        console.log(numeros[0]);
        numeros.push(6);
        console.log(numeros);
        numeros.pop();
        console.log(numeros);

Promises e Async/Await

let promessa = new Promise((resolve, reject) => {
            let sucesso = true;
            if (sucesso) {
                resolve("Deu certo!");
            } else {
                reject("Deu errado!");
            }
        });

        promessa.then((mensagem) => {
            console.log(mensagem);
        }).catch((erro) => {
            console.log(erro);

        async function asyncFunction() {
            try {
                let resultado = await promessa;
                console.log(resultado);
            } catch (erro) {
                console.log(erro);
            }
        }

        asyncFunction();

Manipulação de APIs

fetch('https://api.exemplo.com/dados')
            .then(response => response.json())
            .then(data => console.log(data))
            .catch(error => console.error('Erro:', error));

Módulos em JavaScript

        export const nome = "João";
